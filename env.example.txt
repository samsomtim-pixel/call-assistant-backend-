# Twilio Configuration
# Get these from https://console.twilio.com/

# Account SID and Auth Token (from Twilio Console Dashboard)
TWILIO_ACCOUNT_SID=your_account_sid_here
TWILIO_AUTH_TOKEN=your_auth_token_here

# API Key and Secret (create at https://console.twilio.com/us1/develop/api-keys)
# These are used to generate access tokens for the browser
TWILIO_API_KEY=your_api_key_here
TWILIO_API_SECRET=your_api_secret_here

# Your Twilio phone number (E.164 format, e.g., +1234567890)
# This is the number that will appear as the caller ID
TWILIO_PHONE_NUMBER=+1234567890

# TwiML App SID (OPTIONAL - recommended for production)
# Create at https://console.twilio.com/us1/develop/voice/manage/twiml-apps
# Configure the TwiML App's Voice URL to: https://your-backend-url/api/voice
# If not set, the backend will construct the TwiML URL automatically
TWILIO_TWIML_APP_SID=

# Media Stream URL (REQUIRED for audio streaming)
# This is the WebSocket URL where Twilio will send real-time audio
# For local development:
#   1. Install ngrok: https://ngrok.com/
#   2. Run: ngrok http 3001
#   3. Copy the HTTPS URL and set: MEDIA_STREAM_URL=wss://your-ngrok-url.ngrok.io/api/stream
# For production: Use your actual domain with SSL
# Format: wss://your-domain.com/api/stream
MEDIA_STREAM_URL=wss://your-ngrok-url.ngrok.io/api/stream

# TwiML URL (OPTIONAL - auto-generated if not set)
# The URL where Twilio will fetch TwiML instructions for the call
# For local dev with ngrok: https://your-ngrok-url.ngrok.io/api/voice
# For production: https://your-domain.com/api/voice
TWIML_URL=

# Server Port
PORT=3001
